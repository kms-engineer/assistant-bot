 50    def add_contact(self, name: Name, phone: Phone) -> str:
 51        try:
 52            # Try to find existing contact
 53            contact = self.address_book.find(name.value)
 54            try:
 55                contact.add_phone(phone)
 56                return f"Phone number {phone.value} added to existing contact {name.value}."
 57            except ValueError as e:
 58                # Phone already exists for this contact
 59                if "already exists" in str(e):
 60                    return (
 61                        f"Phone number {phone.value} already exists for {name.value}."
 62                    )
 63                raise  # Re-raise if it's a different ValueError
 64        except KeyError:
 65            # Contact doesn't exist, create new one
 66            contact = Contact.create(
 67                name,
 68                lambda: IDGenerator.generate_unique_id(
 69                    lambda: self.address_book.get_ids()
 70                ),
 71            )
 72            contact.add_phone(phone)
 73            self.address_book.add_record(contact)
 74            return f"Contact {name.value} added with phone {phone.value}."
