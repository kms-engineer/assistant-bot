 37
 38    def __init__(self, base: Type[DeclarativeBase], data_dir: Path = None):
 39        self.resolver = DataPathResolver(data_dir) if data_dir else DataPathResolver()
 40        self._is_initialized = False
 41        self._base_class = base
 42        self._session_factory = None
 43        self._engine = None
 44
 45    def _create_session(self) -> Session:
 46        if not self._is_initialized:
 47            raise RuntimeError(
 48                "SQLiteStorage is not initialized. Call 'initialize' method first."
 49            )
 50        if self._session_factory is None:
 51            raise RuntimeError(
 52                "Session factory is not set. Ensure 'initialize' method has been called."
 53            )
 54        return self._session_factory()
 55
 56    def initialize(self, db_name: str) -> None:
 57        db_path = self.resolver.get_full_path(db_name)
 58        print(f"Initializing SQLite database at {db_path}")
 59        self._engine: Engine = create_engine(
 60            f"sqlite:///{db_path}", echo=False, future=True
 61        )
 62        self._session_factory = sessionmaker(bind=self._engine, expire_on_commit=False)
 63        self._is_initialized = True
 64        self._base_class.metadata.create_all(self._engine)
 65
 66    def save_entity(self, entity: T) -> T:
 67        log.debug(f"Saving entity of type {type(entity).__name__}")
 68        with self._create_session() as session:
 69            try:
 70                merged_entity = session.merge(entity)
 71                session.commit()
 72                session.expunge(merged_entity)
 73                return merged_entity
 74            except Exception as e:
 75                log.error(f"Failed to save entity: {e}")
 76                session.rollback()
 77                raise
 78
 79    def get_entity(self, model_class: Type[T], primary_key: Any) -> Optional[T]:
 80        log.debug(f"Getting {model_class.__name__} with pk={primary_key}")
 81        with self._create_session() as session:
 82            entity = session.get(model_class, primary_key)
 83            if entity:
 84                session.expunge(entity)  # '  
 85            return entity
 86
 87    def get_all(self, model_class: Type[T]) -> List[T]:
 88        log.debug(f"Getting all entities for {model_class.__name__}")
 89        with self._create_session() as session:
 90            entities = session.query(model_class).all()
 91            session.expunge_all()  # '  '
 92            return entities
 93
 94    def delete_entity(self, entity: T) -> None:
 95        log.debug(f"Deleting entity of type {type(entity).__name__}")
 96        with self._create_session() as session:
 97            try:
 98                #  '   ,     
 99                session.delete(session.merge(entity))
100                session.commit()
101            except Exception as e:
102                log.error(f"Failed to delete entity: {e}")
103                session.rollback()
104                raise

106    def save(self, data: Any, filename: str, **kwargs) -> str:
107        self.initialize(db_name=filename)
108
109        if isinstance(data, AddressBook):
110            # Clear existing contacts and save new ones
111            self._clear_and_save(
112                DBContact, data.data.values(), ContactMapper.to_dbmodel
113            )
114            return filename
115        elif isinstance(data, Notebook):
116            # Clear existing notes and save new ones
117            self._clear_and_save(DBNote, data.data.values(), NoteMapper.to_dbmodel)
118            return filename
119
120        return (
121            "Unsupported data type for save operation. Supported: AddressBook, Notebook"
122        )
123
124    def _clear_and_save(self, model_class, entities, mapper_func):
125        with self._create_session() as session:
126            try:
127                # Get existing IDs from database
128                existing_ids = {
129                    record.id for record in session.query(model_class.id).all()
130                }
131
132                # Get IDs from entities to save
133                entity_ids = set()
134                entities_list = list(entities)
135
136                # Save/update all entities
137                for entity in entities_list:
138                    db_model = mapper_func(entity)
139                    entity_ids.add(db_model.id)
140                    merged_entity = session.merge(db_model)
141                    session.add(merged_entity)
142
143                # Delete records that are no longer in the collection
144                ids_to_delete = existing_ids - entity_ids
145                if ids_to_delete:
146                    session.query(model_class).filter(
147                        model_class.id.in_(ids_to_delete)
148                    ).delete(synchronize_session=False)
149
150                session.commit()
151            except Exception as e:
152                log.error(f"Failed to clear and save: {e}")
153                session.rollback()
154                raise
155
156    def load(self, filename: str, **kwargs) -> Optional[Any]:
157        if not self._is_initialized:
158            self.initialize(db_name=filename)
159        try:
160            db_contacts = self.get_all(DBContact)
161            address_book = AddressBook()
162            for db_contact in db_contacts:
163                contact = ContactMapper.from_dbmodel(db_contact)
164                address_book.add_record(contact)
165            return address_book
166        except Exception as e:
167            log.error(f"Failed to load address book: {e}")
168            return None
169
170    def load_notes(self, filename: str, **kwargs) -> Optional[Notebook]:
171        if not self._is_initialized:
172            self.initialize(db_name=filename)
173        try:
174            db_notes = self.get_all(DBNote)
175            notebook = Notebook()
176            for db_note in db_notes:
177                note = NoteMapper.from_dbmodel(db_note)
178                notebook[note.id] = note
179            return notebook
180        except Exception as e:
181            log.error(f"Failed to load notebook: {e}")
182            return None
