10    @staticmethod
11    def validate(phone: str) -> Union[str, bool]:
12        if not isinstance(phone, str):
13            return ValidationConfig.PHONE_ERROR_NOT_STRING
14
15        # Phone can start with + or a digit
16        if not phone or len(phone) == 0:
17            return ValidationConfig.PHONE_ERROR_EMPTY
18
19        if phone[0] == "+":
20            phone = phone[1:]
21        if not (all(c.isdigit() for c in phone)):
22            return ValidationConfig.PHONE_ERROR_INVALID_FORMAT
23
24        # Extract digits only (skip the + if present)
25        digits = "".join(c for c in phone if c.isdigit())
26
27        # Check length is between 8 and 15 digits
28        if (
29            len(digits) < ValidationConfig.PHONE_MIN_DIGITS
30            or len(digits) > ValidationConfig.PHONE_MAX_DIGITS
31        ):
32            return ValidationConfig.PHONE_ERROR_INVALID_LENGTH
33
34        return True

37    def normalize(raw: str) -> str:
38        if not raw:
39            return ""
40        if raw.startswith("+"):

42        return re.sub(r"\D+", "", raw)
